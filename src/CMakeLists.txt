	QT4_CREATE_TRANSLATION(QM_FILES ${lang_SOURCES} ${TRANSLATIONS_FILES})
else (UPDATE_TRANSLATIONS)
	QT4_ADD_TRANSLATION(QM_FILES ${TRANSLATIONS_FILES})
